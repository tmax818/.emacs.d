** org-todo keywords



#+begin_src emacs-lisp
  (setq org-todo-keywords
      '((sequence "APPT" "STUFF" "NEXT" "SOMEDAY" "PROJECT" "REFERENCE" "|" "DONE" "CANCELED")))

  (setq org-todo-keyword-faces
      '(
	("APPT" . "red")
	("STUFF" . "orange")
	("NEXT" . org-warning)
	("SOMEDAY" . "yellow")
	("PROJECT" . "green")
	("REFERENCE" . "blue")
	("DONE" . "red")
	("CANCELED" . (:foreground "blue" :weight bold))
	))

#+end_src
